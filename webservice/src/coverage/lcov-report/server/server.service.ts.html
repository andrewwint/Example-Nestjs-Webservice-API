import { Model } from 'mongoose';
import { Injectable, Logger, NotFoundException } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { CreateServerDTO } from './dto/create-server.dto';
import { Server } from './server.interface';
import { UpdateServerDTO } from './dto/update-server.dto';
import * as util from 'util';
import { exec } from 'child_process';
const exec_async = util.promisify(exec);
import { AwsService } from '../aws/aws.service';
 
@Injectable()
export class ServerService {
  private logger = new Logger('server');
  public shell_command: String;
 
  constructor(
    @InjectModel('Server') private readonly serverModel: Model<Server>,
    private readonly awsService: AwsService,
  ) {}
 
  async getServers(): Promise<Server[]> {
    const result = await this.serverModel.find().exec();
    this.logger.log(`Found ${result.length} servers`);
    return result;
  }
 
  async getServersByStackName(stackname: String): Promise<Server[]> {
    const result = await this.serverModel.find({ stackname: stackname }).exec();
    this.logger.log(`Found ${result.length} servers for stack ${stackname}`);
    return result;
  }
 
  async setServer(createServerDTO: CreateServerDTO): Promise<Server> {
    const newServer = await new this.serverModel(createServerDTO).save();
    this.logger.log(
      `Created server ${newServer.shortname} with id: ${newServer._id}`,
    );
    return newServer;
  }
 
  async setServersByStackName(stackname: string): Promise<Server[]> {
    this.logger.log(`Refreshing or adding new stack for ${stackname}`);
 
    const instances = (await this.awsService.describeInstance(stackname))
      .Reservations;
 
    if (instances.length > 0) {
      await this.deleteServerByStackName(stackname);
    }
 
    const servers = [];
    for (const data of instances) {
      for (const instance of data.Instances) {
        let server = this.setAWSDescribe2ServerObj(instance, stackname);
        servers.push(server);
        await this.setServer(server);
      }
    }
    return servers;
  }
 
  async getServerByShortName(shortname: String): Promise<Server> {
    const result = await this.serverModel
      .findOne({ shortname: shortname })
      .exec();
    this.logger.log(
      `Found server details for: "${result.shortname}" by shortname`,
    );
    return result;
  }
 
  async getServerByInstanceId(instance_id: String): Promise<Server> {
    const result = await this.serverModel
      .findOne({ instance_id: instance_id })
      .exec();
    this.logger.log(
      `Found server details for: "${result.instance_id}" by instance_id`,
    );
    return result;
  }
 
  async getServersByType(type: String): Promise<Server[]> {
    const results = await this.serverModel.find({ type: type }).exec();
    this.logger.log(`Found ${results.length} server(s) for type ${type} `);
    return results;
  }
 
  async updateServerByShortName(
    shortname: String,
    updateServerDTO: UpdateServerDTO,
  ): Promise<Server> {
    const result = await this.serverModel.findOneAndUpdate(
      { shortname: shortname },
      updateServerDTO,
      { new: true },
    );
    this.logger.log(`Updated ${shortname}`);
    return result;
  }
 
  async deleteServerByShortName(shortname: String): Promise<Server> {
    const result = await this.serverModel.findOneAndDelete(
      { shortname: shortname },
      { select: '_id' },
    );
    this.logger.log(`Removed ${shortname} with id: ${result._id}`);
    return result;
  }
 
  async deleteServerByStackName(stackname: String): Promise<String> {
    const result = await this.serverModel.deleteMany({ stackname: stackname });
    let message = `Removed ${result.deletedCount} instances with stackName ${stackname} `;
    this.logger.log(message);
    return message;
  }
 
  async installApplicationByStackName(
    stackname: String,
    nodesize: String,
  ): Promise<String> {
    const instances = await this.getServersByStackName(stackname);
    for (const instance of instances) {
      await this.installApplicationByInstanceId(
        instance.instance_id,
        instance.environment,
        nodesize,
        instance,
      );
    }
    return '';
  }
 
  async installApplicationByInstanceId(
    instance_id: String,
    environment: String,
    nodesize: String,
    instanceobj = undefined,
  ): Promise<String> {
    let server: Server;
 
    if (instanceobj == undefined) {
      server = await this.getServerByInstanceId(instance_id);
    } else {
      server = instanceobj;
    }
 
    await this.setInstallShellCmdByServerObj(server, environment, nodesize);
 
    try {
      const { stdout, stderr } = await exec_async(
        `ssh -o StrictHostKeyChecking=no -i ${process.env.AWS_PEM_FILE} ${this.shell_command}`,
      );
      this.logger.log(
        `Installing ${environment} using ${
          this.shell_command
        } status: ${stdout.trim()}`,
      );
      return `${stdout}`;
    } catch (error) {
      throw new NotFoundException(
        this.logger.error(
          `Tried to install ${environment} using ${this.shell_command} `,
          error.stderr,
        ),
      );
    }
  }
 
  async setInstallShellCmdByServerObj(
    server: Server,
    environment: String,
    nodesize: String,
  ): Promise<String> {
    this.logger.log(
      `Composing ssh command for ${environment} ServerType: ${server.servertype}, NodeSize: ${nodesize} `,
    );
    this.shell_command = `-t ${server.username}@${server.ip} install ${environment} ${server.servertype} ${nodesize} ${server.version}`;
    return this.shell_command;
  }
 
  setAWSDescribe2ServerObj(
    awsInstance: any,
    stackName: String,
  ): CreateServerDTO {
    this.logger.log(
      `Mutating AWS describeInstances output for Stack ${stackName} to Model<Server>`,
    );
    const server: CreateServerDTO = {
      shortname: awsInstance.InstanceId.trim(),
      stackname: stackName.trim(),
      username: 'ubuntu',
      instance_id: awsInstance.InstanceId.trim(),
      name: awsInstance.InstanceId,
      version: awsInstance.Tags.filter(tag => tag.Key == 'Version')[0].Value,
      environment: awsInstance.Tags.filter(tag => tag.Key == 'environment')[0]
        .Value,
      ip: awsInstance.PublicIpAddress.trim(),
      private_ip: awsInstance.PrivateIpAddress.trim(),
      servertype: awsInstance.Tags.filter(
        tag => tag.Key == 'ServerType',
      )[0].Value.split(' ')[0],
      stacktype: awsInstance.Tags.filter(tag => tag.Key == 'StackType')[0]
        .Value,
      status: `Code ${awsInstance.State.Code} Name ${awsInstance.State.Name}`,
    };
 
    return server;
  }
}
