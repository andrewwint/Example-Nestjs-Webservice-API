import { Controller, Get, Param, Body, Post, Logger, Ip } from '@nestjs/common';
import { XmlimportService } from './xmlimport.service';
import { XmlFileImportDTO } from './dto/xmlfileimport.dto';
import { ApiTags } from '@nestjs/swagger';
 
@ApiTags('xmlimport')
@Controller('xmlimport')
export class XmlimportController {
  private logger = new Logger('xmlimport');
 
  constructor(private readonly xmlimportService: XmlimportService) {}
 
  @Get('/:id')
  importXmlByContentId(
    @Param('id') id: string,
    @Ip() ip: string,
  ): Promise<String> {
    this.logger.log(`Importing from Content Id "${id}" from ${ip}`);
    return this.xmlimportService.importXmlByContentId(id);
  }
 
  @Post('/xml')
  importXml(@Body() xmlFile: XmlFileImportDTO): Promise<String> {
    this.logger.log(`Importing Single XML file "${xmlFile}"`);
    return this.xmlimportService.importXml(xmlFile);
  }
 
  @Post('/batchxml')
  importBatchXml(@Body() xmlFileImportDTO: XmlFileImportDTO): Promise<String> {
    this.logger.log(`Importing Batch from "${xmlFileImportDTO}"`);
    return this.xmlimportService.importBatchXml(xmlFileImportDTO);
  }
}
